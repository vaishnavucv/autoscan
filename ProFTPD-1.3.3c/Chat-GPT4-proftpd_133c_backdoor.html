To achieve your objectives, you can use Python's subprocess module to run nmap commands and then parse the output with regular expressions using the re module. The script below outlines a basic approach to meet your requirements:

  1. The script accepts an IP address and a scan report name as inputs.
  2. It executes the specified nmap command using the subprocess module.

Here's a sample Python script that does this:

python
import subprocess
import re
import sys

def run_nmap_scan(ip_address, scan_name):
    # Constructing the nmap command
    command = f"nmap -p21 -vv -sV -oN {scan_name}.txt {ip_address}"
    
    try:
        # Running the nmap command
        output = subprocess.check_output(command, shell=True, stderr=subprocess.STDOUT, text=True)
        
        # Print the full output (Optional)
        print("Nmap scan output:\n", output)
        
        # Using regex to find port status
        pattern = re.compile(r"21/tcp\s+(\w+)\s+(\S+)\s+syn-ack ttl \d+ (\S+ \d+\.\d+\.\d+)")
        matches = pattern.search(output)
        
        if matches:
            status, service, version = matches.groups()
            print(f"Port 21 status: {status}, Service: {service}, Version: {version}")
        else:
            print("Port 21 details not found.")
            
    except subprocess.CalledProcessError as e:
        print("Error during scan:", e.output)

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: python script.py <IP Address> <Scan Report Name>")
    else:
        ip_address = sys.argv[1]
        scan_name = sys.argv[2]
        run_nmap_scan(ip_address, scan_name)
